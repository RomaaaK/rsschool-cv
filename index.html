            function humanReadable(seconds) {
                let hh = String('0' + Math.trunc(seconds / 60 / 60)).slice(-2);
                let mm = String('0' + Math.trunc(seconds / 60 % 60)).slice(-2);
                let ss = String('0' + Math.trunc(seconds % 60 % 60)).slice(-2);
                return `${hh}:${mm}:${ss}`;
                }
